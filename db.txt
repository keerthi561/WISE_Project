CREATE TABLE management(managementId int, managementName varchar(255), managementDesignation varchar(255), managementUserId varchar(255), managementPassword varchar(255), managementMobile varchar(255), homeAddress varchar(255), pincode varchar(255), city varchar(255), district varchar(255), state varchar(255), country varchar(255), managementImage varchar(255), managementGender varchar(255), managementFb varchar(255), managementTwitter varchar(255), managementLinkedIn varchar(255), PRIMARY KEY (managementId));
